[C#]
public static void TXSync_LockPianificazioni(TipoPianificazione tipoPianificazione) { 
    Dao dao = new Dao(); 
    string sql = ""; 
    sql = "UPDATE Pianificazioni SET " 
        + "SyncStato = '" + ConfigurationManager.AppSettings["SYNC_STATO_IN_USO"] + "' " 
        + "WHERE SyncStato = '" + ConfigurationManager.AppSettings["SYNC_STATO_DA_SINCRONIZZARE"] + "' " 
        + "AND SyncTask IS NOT NULL "; 
    if (tipoPianificazione == TipoPianificazione.VIAGGIO) { 
        sql += "AND IdSpedizione IS NULL"; 
    } else if (tipoPianificazione == TipoPianificazione.SPEDIZIONE) { 
        sql += "AND IdSpedizione IS NOT NULL"; 
    } 
    dao.ExecuteNonQuery(sql); 
}
