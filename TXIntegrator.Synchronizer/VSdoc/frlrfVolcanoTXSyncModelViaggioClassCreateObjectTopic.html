[C#]
private void CreateObject(DataRow viaggio) { 
    this.Id = (viaggio["Id"] != DBNull.Value) ? (int)viaggio["Id"] : 0; 
    this.KeyViaggio = (viaggio["KeyViaggio"] != DBNull.Value) ? (string)viaggio["KeyViaggio"] : ""; 
    this.Societa = (viaggio["Societa"] != DBNull.Value) ? (string)viaggio["Societa"] : ""; 
    this.Filiale = (viaggio["Filiale"] != DBNull.Value) ? (string)viaggio["Filiale"] : ""; 
    this.Anno = (viaggio["Anno"] != DBNull.Value) ? Convert.ToInt32(viaggio["Anno"]) : 0; 
    this.Progressivo = (viaggio["Progressivo"] != DBNull.Value) ? Convert.ToInt32(viaggio["Progressivo"]) : 0; 
    this.DataViaggio = (viaggio["DataViaggio"] != DBNull.Value) ? DateTime.ParseExact(viaggio["DataViaggio"].ToString(), ConfigurationManager.AppSettings["DB_DATE_PATTERN"], null) : DateTime.MinValue; 
    this.Reparto = (viaggio["Reparto"] != DBNull.Value) ? (string)viaggio["Reparto"] : ""; 
    this.Servizio = (viaggio["Servizio"] != DBNull.Value) ? (string)viaggio["Servizio"] : ""; 
    this.CodiceMezzo = (viaggio["CodiceMezzo"] != DBNull.Value) ? viaggio["CodiceMezzo"].ToString().Trim() : ""; 
    this.CodiceAutista = (viaggio["CodiceAutista"] != DBNull.Value) ? viaggio["CodiceAutista"].ToString().Trim() : ""; 
    this.DestinazioneViaggio = (viaggio["DestinazioneViaggio"] != DBNull.Value) ? (string)viaggio["DestinazioneViaggio"] : ""; 
    this.Note = (viaggio["Note"] != DBNull.Value) ? (string)viaggio["Note"] : ""; 
    this.UtenteCompetenza = (viaggio["UtenteCompetenza"] != DBNull.Value) ? (string)viaggio["UtenteCompetenza"] : ""; 
    this.MailUtenteCompetenza = (viaggio["MailUtenteCompetenza"] != DBNull.Value) ? (string)viaggio["MailUtenteCompetenza"] : ""; 
}
