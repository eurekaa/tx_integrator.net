[C#]
public static void DBSyncSpedizioni() { 
    try { 
        List<Pianificazione> spedizioniPianificate = Pianificazione.DBSync_GetPianificazioni(TipoPianificazione.SPEDIZIONE); 
        Spedizione spedizione = null; 
        for (int i = 0; i < spedizioniPianificate.Count; i++) { 
            spedizione = new Spedizione(spedizioniPianificate[i].IdViaggio, spedizioniPianificate[i].IdSpedizione); 
            Evento evento = spedizione.TXGetStatus(login); 
            if (spedizioniPianificate[i].Stato != evento.Stato) { 
                spedizioniPianificate[i].Stato = evento.Stato; 
                spedizioniPianificate[i].DBUpdate(); 
                evento.IdPianificazione = spedizioniPianificate[i].Id; 
                evento.DBInsert(); 
            } 
        } 
    } catch (Exception ex) { 
        log.Error(ex.Message, ex); 
    } 
}
