[C#]
private void CreateObject(DataRow pianificazione) { 
    this.Id = (Int32)pianificazione["Id"]; 
    this.IdViaggio = (pianificazione["IdViaggio"] != DBNull.Value) ? (int)pianificazione["IdViaggio"] : 0; 
    this.IdSpedizione = (pianificazione["IdSpedizione"] != DBNull.Value) ? (int)pianificazione["IdSpedizione"] : 0; 
    this.Stato = (pianificazione["Stato"] != DBNull.Value) ? (string)pianificazione["Stato"] : ""; 
    this.SyncStato = (pianificazione["SyncStato"] != DBNull.Value) ? (string)pianificazione["SyncStato"] : ""; 
    this.SyncTask = (pianificazione["SyncTask"] != DBNull.Value) ? (string)pianificazione["SyncTask"] : ""; 
}
