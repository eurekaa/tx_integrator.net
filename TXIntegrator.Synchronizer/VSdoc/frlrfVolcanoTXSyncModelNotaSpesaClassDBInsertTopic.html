[C#]
public void DBInsert() { 
    Dao dao = new Dao(); 
    string sql = ""; 
    sql += "INSERT INTO NotaSpese "; 
    sql += "(IdViaggio, Data, Tipo, Codice, Descrizione, Indirizzo, GeoCoordinate, Prezzo, Valuta) "; 
    sql += "VALUES (@IdViaggio, "; 
    sql += (this.Data != DateTime.MinValue) ? "@Data, " : "NULL, "; 
    sql += "@Tipo, @Codice, @Descrizione, @Indirizzo, @GeoCoordinate, @Prezzo, @Valuta)";             
 
    Hashtable parameters = new Hashtable(); 
    parameters.Add("@IdViaggio", this.IdViaggio);             
    if(this.Data!=DateTime.MinValue){ 
        parameters.Add("@Data", this.Data); 
    } 
    parameters.Add("@Tipo", this.Tipo); 
    parameters.Add("@Codice", this.Codice); 
    parameters.Add("@Descrizione", this.Descrizione); 
    parameters.Add("@Indirizzo", this.Indirizzo); 
    parameters.Add("@GeoCoordinate", this.GeoCoordinate); 
    parameters.Add("@GeoCoordinate", this.Prezzo); 
    parameters.Add("@Valuta", this.Valuta); 
 
    dao.ExecuteNonQuery(sql, parameters); 
}
