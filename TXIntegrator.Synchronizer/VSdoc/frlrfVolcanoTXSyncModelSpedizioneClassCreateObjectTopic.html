[C#]
private void CreateObject(int id_viaggio, DataRow spedizione) { 
    this.Id = (int)spedizione["Id"]; 
    this.IdViaggio = id_viaggio; 
    this.KeySpedizione = (spedizione["KeySpedizione"] != DBNull.Value) ? (string)spedizione["KeySpedizione"] : ""; 
    this.Tipo = (spedizione["Tipo"] != DBNull.Value) ? (string)spedizione["Tipo"] : ""; 
    this.Societa = (spedizione["Societa"] != DBNull.Value) ? (string)spedizione["Societa"] : ""; 
    this.Filiale = (spedizione["Filiale"] != DBNull.Value) ? (string)spedizione["Filiale"] : ""; 
    this.Anno = (spedizione["Anno"] != DBNull.Value) ? Convert.ToInt32(spedizione["Anno"]) : 0; 
    this.Progressivo = (spedizione["Progressivo"] != DBNull.Value) ? Convert.ToInt32(spedizione["Progressivo"]) : 0; 
    this.Servizio = (spedizione["Servizio"] != DBNull.Value) ? (string)spedizione["Servizio"] : ""; 
    this.Reparto = (spedizione["Servizio"] != DBNull.Value) ? (string)spedizione["Servizio"] : ""; 
    this.MittenteRagSoc = (spedizione["MittenteRagSoc"] != DBNull.Value) ? (string)spedizione["MittenteRagSoc"] : ""; 
    this.DestinazioneRagSoc = (spedizione["DestinazioneRagSoc"] != DBNull.Value) ? (string)spedizione["DestinazioneRagSoc"] : ""; 
    this.DestinazioneIndirizzo = (spedizione["DestinazioneIndirizzo"] != DBNull.Value) ? (string)spedizione["DestinazioneIndirizzo"] : ""; 
    this.DestinazioneCap = (spedizione["DestinazioneCap"] != DBNull.Value) ? (string)spedizione["DestinazioneCap"] : ""; 
    this.DestinazioneLocalita = (spedizione["DestinazioneLocalita"] != DBNull.Value) ? (string)spedizione["DestinazioneLocalita"] : ""; 
    this.DestinazioneProvincia = (spedizione["DestinazioneProvincia"] != DBNull.Value) ? (string)spedizione["DestinazioneProvincia"] : ""; 
    this.DestinazioneNazione = (spedizione["DestinazioneNazione"] != DBNull.Value) ? (string)spedizione["DestinazioneNazione"] : ""; 
    this.DestinazioneGeoLoc = (spedizione["DestinazioneGeoLoc"] != DBNull.Value) ? (string)spedizione["DestinazioneGeoLoc"] : ""; 
    this.Colli = (spedizione["Colli"] != DBNull.Value) ? Convert.ToInt32(spedizione["Colli"]) : 0; 
    this.Peso = (spedizione["Peso"] != DBNull.Value) ? Convert.ToDouble(spedizione["Peso"]) : 0; 
    this.Volume = (spedizione["Volume"] != DBNull.Value) ? Convert.ToDouble(spedizione["Volume"]) : 0; 
    this.Note = (spedizione["Note"] != DBNull.Value) ? (string)spedizione["Note"] : ""; 
 
    // se la spedizione non è geo-localizzata la geolocalizzo 
    if (this.DestinazioneGeoLoc == "") { 
        this.GeoLocalizzaSpedizione(); 
    } 
}
