public Spedizione(int id_viaggio, int id_spedizione) { 
    Dao dao = new Dao(); 
    string sql = "SELECT * FROM Spedizioni WHERE Id = " + id_spedizione; 
    DataSet data = dao.ExecuteQuery(sql); 
    DataRow spedizione = data.Tables[0].Rows[0]; 
    CreateObject(id_viaggio, spedizione); 
}
