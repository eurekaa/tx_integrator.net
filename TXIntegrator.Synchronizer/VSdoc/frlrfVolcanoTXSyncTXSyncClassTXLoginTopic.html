[C#]
public static void TXLogin() { 
    login = new Login(); 
    login.Dispatcher = ConfigurationManager.AppSettings["LOGIN_DISPATCHER"]; 
    login.Password = ConfigurationManager.AppSettings["LOGIN_PASSWORD"]; 
    login.SystemNr = Convert.ToInt32(ConfigurationManager.AppSettings["LOGIN_SYSTEMNR"]); 
    login.Integrator = ConfigurationManager.AppSettings["LOGIN_INTEGRATOR"]; 
    login.Language = ConfigurationManager.AppSettings["LOGIN_LANGUAGE"]; 
}
