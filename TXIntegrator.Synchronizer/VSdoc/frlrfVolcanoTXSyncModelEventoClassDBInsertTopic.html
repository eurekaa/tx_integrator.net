[C#]
public void DBInsert() { 
    Dao dao = new Dao(); 
    string sql = ""; 
    sql += "INSERT INTO Eventi "; 
    sql += "(IdPianificazione, Stato, Data, SyncData, SyncTipo, SyncTask, SyncStato, XmlRequest, XmlResponse, Note) "; 
    sql += "VALUES (@IdPianificazione, @Stato, "; 
    sql += (this.Data != DateTime.MinValue) ? "@Data, " : "NULL, "; 
    sql += "@SyncData, @SyncTipo, @SyncTask, @SyncStato, @XmlRequest, @XmlResponse, "; 
    sql += (this.Note != "") ? "@Note)" : "NULL)"; 
 
    Hashtable parameters = new Hashtable(); 
    parameters.Add("@IdPianificazione", this.IdPianificazione); 
    parameters.Add("@Stato", this.Stato); 
    if(this.Data!=DateTime.MinValue){ 
        parameters.Add("@Data", this.Data); 
    } 
    parameters.Add("@SyncData", this.SyncData); 
    parameters.Add("@SyncTipo", this.SyncTipo); 
    parameters.Add("@SyncTask", this.SyncTask); 
    parameters.Add("@SyncStato", this.SyncStato); 
    parameters.Add("@XmlRequest", this.XmlRequest); 
    parameters.Add("@XmlResponse", this.XmlResponse); 
    if(this.Note!=""){ 
        parameters.Add("@Note", this.Note); 
    } 
    dao.ExecuteNonQuery(sql, parameters); 
}
